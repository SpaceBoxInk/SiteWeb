120 void MTerrain::loadCouche(std::string const & fichier, MTypeCouche const & type)
121 {
122  std::ifstream fichierCouche(fichier, std::ios::binary);
123  if (fichierCouche)
124  {
125  uint32_t nbElemRequis = (unsigned int)(taille.getX() * taille.getY());
126  uint8_t ID;
127  std::vector<uint8_t> vecteurControle;
128  vecteurControle.insert(vecteurControle.begin(),
129  std::istream_iterator<uint8_t>(fichierCouche),
130  std::istream_iterator<uint8_t>());
131  if (vecteurControle.size() == nbElemRequis)
132  {
133  for (unsigned int i = 0; i < vecteurControle.size(); ++i)
134  {
135  ID = vecteurControle[i];
136  if (ID == 0 && type != MTypeCouche::SOL)
137  {
138  // le rien
139  tuiles[i]->deletePartieCouche(type);
140  continue;
141  }
142  else if (ID == 0 && type == MTypeCouche::SOL)
143  {
144  throw std::invalid_argument(
145  std::string("Id 0 not expected for couche SOL in offset ") + std::to_string(i));
146  }
147  MPartieCouche const & couche = getTypeList(type).at(ID);
148  // si on veut mettre une couche à une tuile pas existante
149  // il faut aussi que le type soit le sol, on commence par le sol !
150  if (tuiles.size() <= i && type == MTypeCouche::SOL)
151  {
152  // SEE : push_back may be replaced by something related to the position "i"
153  tuiles.push_back(

155  couche.getPlaceDispo()));
